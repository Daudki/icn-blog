Packet Sniffer & Analyzer

Intermediate

Capture a small number of packets and identify whether they are TCP or UDP.

packet_sniffer.py
from scapy.all import sniff, IP, TCP, UDP

def analyze(packet):
    if packet.haslayer(IP):
        if packet.haslayer(TCP):
            print('TCP', packet[IP].src, '->', packet[IP].dst)
        elif packet.haslayer(UDP):
            print('UDP', packet[IP].src, '->', packet[IP].dst)

sniff(count=5, prn=analyze)
